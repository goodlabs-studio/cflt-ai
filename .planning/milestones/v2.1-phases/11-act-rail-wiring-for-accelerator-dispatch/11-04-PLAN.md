    ```python
    # Pre-flight profile gate (Phase 11 — Plan 11-04). Skipped when profile_name=None
    # (back-compat with Plan 11-02 tests). When supplied, fail-closed BEFORE any
    # kustomize/oc invocation or per-layer ACTA-04 emission.
    artifact_id = artifact.get("id", "")
    if profile_name is not None:
        try:
            profile = load_profile(profile_name, customer=args.get("overlay"))
        except ValueError as exc:
            return ExecutionResult(
                status="refused",
                duration_seconds=0.0,
                stderr_tail=f"Unknown profile {profile_name!r}: {exc}",
            )
        if not check_profile_permits(profile, artifact_id):
            # Single ACTA-04 entry (blocked-by-profile) — no layer_id, no per-layer entries.
            emit_activity_log_apply(
                overlay=args.get("overlay", "base"),
                plan_path=args.get("plan_path", ""),
                artifact_id=artifact_id,
                profile_name=profile_name,
                confirmation_status="blocked",
                execution_result="refused",
                duration_seconds=0.0,
                gate_results=args.get("gate_results", []),
                operator=args.get("operator", "unknown"),
            )
            return ExecutionResult(
                status="refused",
                duration_seconds=0.0,
                stderr_tail=f"Profile {profile_name!r} does not permit operation on {artifact_id!r}",
            )
    ```
